 /*--- 记录NSLOG ---*/
[JLLogManager setLog:true IsMore:false Level:JLLOG_COMPLETE];
[JLLogManager saveLogAsFile:true];
[JLLogManager logWithTimestamp:true];
[JLLogManager clearLog];
